cmake_minimum_required(VERSION 3.20)
project(VoxelEngine)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)


# 2. Find the Vulkan SDK
# This sets variables like Vulkan_INCLUDE_DIRS and Vulkan_LIBRARIES
find_package(Vulkan REQUIRED)

function(compile_shader TARGET SHADER)
    get_filename_component(FILE_NAME ${SHADER} NAME)
    set(SPIRV "${CMAKE_CURRENT_BINARY_DIR}/${FILE_NAME}.spv")

    add_custom_command(
        OUTPUT ${SPIRV}
        COMMAND Vulkan::glslc ${CMAKE_CURRENT_SOURCE_DIR}/${SHADER} -o ${SPIRV}
        DEPENDS ${SHADER}
        COMMENT "Compiling shader ${SHADER}"
        VERBATIM
    )

    target_sources(${TARGET} PRIVATE ${SPIRV})
endfunction()

option(ENABLE_ASAN "Enable AddressSanitizer")

if(ENABLE_ASAN)
    if(MSVC)
        add_compile_options(/fsanitize=address)
        add_link_options(/fsanitize=address)
    else()
        add_compile_options(-fsanitize=address -fno-omit-frame-pointer -g)
        add_link_options(-fsanitize=address)
    endif()
endif()

add_subdirectory("3rdparty/spdlog")
add_subdirectory("3rdparty/glm")

add_executable(VoxelEngine
    "main.cpp"
    "Engine/Graphics/GraphicsBackends/Vulkan.cpp"
    "Engine/Graphics/GraphicsBackends/Vulkan/VulkanBuffer.cpp"
    "Engine/Graphics/GraphicsBackends/Vulkan/VulkanFence.cpp"
    "Engine/Graphics/GraphicsBackends/Vulkan/VulkanPipeline.cpp"
    "Engine/Graphics/GraphicsBackends/Vulkan/VulkanTexture.cpp"
    "Engine/Graphics/GraphicsBackends/Vulkan/VulkanCommandBuffer.cpp"
    "Engine/Graphics/GraphicsBackends/Vulkan/VulkanFrameBuffer.cpp"
    "Engine/Graphics/GraphicsBackends/Vulkan/VulkanBindingGroups.cpp"
    "Engine/Graphics/GraphicsBackends/GraphicsBackends.cpp"
 "Engine/Graphics/TestRenderer.cpp")
  
if (WIN32)
    target_compile_definitions(VoxelEngine PRIVATE VK_USE_PLATFORM_WIN32_KHR)
    target_compile_definitions(VoxelEngine PRIVATE PLATFORM_WINDOWS)
elseif (ANDROID)
    target_compile_definitions(VoxelEngine PRIVATE VK_USE_PLATFORM_ANDROID_KHR)
    target_compile_definitions(VoxelEngine PRIVATE PLATFORM_ANDROID)
endif()

if (MSVC)
    target_compile_options(VoxelEngine PRIVATE
        $<$<CONFIG:Debug>:
            /permissive-
            /sdl
            /analyze
            /we4700
            /RTC1
        >
    )
endif()

target_link_libraries(VoxelEngine PRIVATE spdlog::spdlog $<$<BOOL:${MINGW}>:ws2_32>)
target_link_libraries(VoxelEngine PRIVATE glm::glm)

# 4. Link Vulkan to your project
# Use the modern imported target Vulkan::Vulkan
target_link_libraries(VoxelEngine PRIVATE Vulkan::Vulkan)
compile_shader(VoxelEngine "test_triangle.frag")
compile_shader(VoxelEngine "test_triangle.vert")
compile_shader(VoxelEngine "test_cube.frag")
